\subsection{Query Execution}

Out of 25 Competency Questions, 10 could be fully answered via SPARQL queries, 6 partially, and 9 could not be answered. Below are representative SPARQL queries for the answerable CQs.\\

\noindent\textbf{CQ 1.1 -- All ski resorts in Trentino:}
\begin{verbatim}
PREFIX ski: <http://skiresorts-trentino.2026.kg/ontology#>
SELECT ?resort ?name WHERE {
  ?resort a ski:SkiResort ; ski:name ?name .
} ORDER BY ?name
\end{verbatim}

\noindent\textbf{CQ 1.3 -- Restaurants near a specific resort:}
\begin{verbatim}
PREFIX ski: <http://skiresorts-trentino.2026.kg/ontology#>
SELECT ?restaurant ?name ?cuisine WHERE {
  ?restaurant a ski:Restaurant ;
    ski:name ?name ; ski:nearResort ?resort .
  ?resort ski:name "Madonna di Campiglio" .
  OPTIONAL { ?restaurant ski:cuisine ?cuisine . }
} ORDER BY ?name
\end{verbatim}

\noindent\textbf{CQ 2.2 -- Resorts with easy slopes:}
\begin{verbatim}
PREFIX ski: <http://skiresorts-trentino.2026.kg/ontology#>
SELECT ?resortName (COUNT(?slope) AS ?easySlopes) WHERE {
  ?slope a ski:SkiSlope ; ski:difficulty "easy" ;
    ski:nearResort ?resort .
  ?resort ski:name ?resortName .
} GROUP BY ?resortName ORDER BY DESC(?easySlopes)
\end{verbatim}

\noindent\textbf{CQ 3.2 -- Resort with most slopes:}
\begin{verbatim}
PREFIX ski: <http://skiresorts-trentino.2026.kg/ontology#>
SELECT ?resortName (COUNT(?slope) AS ?totalSlopes) WHERE {
  ?slope a ski:SkiSlope ; ski:nearResort ?resort .
  ?resort ski:name ?resortName .
} GROUP BY ?resortName ORDER BY DESC(?totalSlopes) LIMIT 10
\end{verbatim}

\noindent\textbf{CQ 5.1 -- Resorts with gondola lifts:}
\begin{verbatim}
PREFIX ski: <http://skiresorts-trentino.2026.kg/ontology#>
SELECT DISTINCT ?resortName WHERE {
  ?lift a ski:SkiLift ; ski:liftType "gondola" ;
    ski:nearResort ?resort .
  ?resort ski:name ?resortName .
} ORDER BY ?resortName
\end{verbatim}
